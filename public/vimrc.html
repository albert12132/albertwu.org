215 " Fugitive
216 nnoremap <leader>gs     :Gstatus<CR>
217 nnoremap <leader>gc     :Gcommit -m "
218 nnoremap <leader>gg     :Git push
219 function! FugitiveStatus()
220     let branch = fugitive#head()
221     if !empty(branch)
222         let branch = " [" . branch . "] "
223     endif
224     return branch
225 endfunction

305 "--------------------------------------------------------------------"
306 " "Syntax Highlighting utility"                                      "
307 "--------------------------------------------------------------------"
308 function! StatuslineCurrentHighlight()
309     let name = synIDattr(synID(line('.'),col('.'),1),'name')
310     if name == ''
311         return ''
312     else
313         return '[' . name . ']'
314     endif
315 endfunction
